Kitty KeyMap

五月 10, 2024

tab

map kitty_mod+enter     new_tab
map kitty_mod+n         next_tab
map kitty_mod+p         previous_tab
map kitty_mod+s         select_tab

window

# Create a new window splitting the space used by the existing one so that
# the two windows are placed side by side if the existing window is wide or
# one above the other if the existing window is tall.
map alt+enter launch --location=split

# 移动活动窗口位置
map alt+shift+k move_window up
map alt+shift+h move_window left
map alt+shift+l move_window right
map alt+shift+j move_window down

# 切换窗口焦点
map alt+h neighboring_window left
map alt+l neighboring_window right
map alt+k neighboring_window up
map alt+j neighboring_window down

# resizeing windows
map alt+left    resize_window narrower  3
map alt+right   resize_window wider     3
map alt+up      resize_window taller    3
map alt+down    resize_window shorter   3

layout

enabled_layouts splits, tall, fat, stack

map kitty_mod+l next_layout

# 切换当前窗口为最大化
map alt+f toggle_layout stack
