Python 装饰器

七月 8, 2024

Python 文档中对于装饰器的解释, 在函数定义中有相关内容

def decorator_function(cnf):
    print(cnf)
    def config_fun(original_function):
        def wrapper_function(*args, **kwargs):
            # 在调用原始函数之前的操作
            print(f"Wrapper executed this before {original_function.__name__}()")

            # 调用原始函数
            ret = original_function(*args, **kwargs)

            # 在调用原始函数之后的操作
            print(f"Wrapper executed this after {original_function.__name__}()")
            return ret

        return wrapper_function
    return config_fun

@decorator_function({'a': 1, 'b': 2})
def display():
    print("Display function ran")

# display = decorator_function({'a': 1, 'b':2})(display)
display()
